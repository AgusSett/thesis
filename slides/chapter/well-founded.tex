\begin{frame}{Relación bien fundada}
	\begin{block}{Definición (Relación bien fundada)}
		Una relación está bien fundada cuando todas sus cadenas descendentes
		\[ x_1 < x_2 < \dots < x_n \]
		tienen un largo acotado.
	\end{block}
	
	\pause
	\begin{block}{Definición (Accesibilidad)}
		Un elemento $x: A$ es accesible, si todo elemento $y < x$ es también accesible.
		\ExecuteMetaData[code/eval.tex]{acc}
	\end{block}
\end{frame}
